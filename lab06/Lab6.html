function LegendrePolynomial(deg)
    if deg == 0
        return Polynomial([1]) #P_0(x) = 1
    end
    
    x = Polynomial([0,1]) #P_1(x) = x
    
    if deg == 1
        return x
    end
    
    return ((2*(deg - 1) + 1)/((deg - 1) + 1)*x*LegendrePolynomial(deg - 1)) - ((deg - 1)/((deg - 1) + 1)*LegendrePolynomial(deg - 2))
end

(x, w) = gauss([T,] N)¶

returns a pair of N quadrature points x[i] and their weights w[i] (i = 1,...,N) to integrate functions on interval (-1, 1), i.e. sum(w .* f.(x)) approximates the integral of f(x) on (-1, 1). T is an optional parameter specifying the floating-point type, default Float64

function test(deg)
    poly_roots = sort([round.(x, digits = 10) for x in roots(LegendrePolynomial(deg))])
    gauss_points = sort([round.(x, digits = 10) for x in gauss(Float64, deg)[1]])
    return poly_roots == gauss_points ?  (true, "Same points") : (false, "Different points")
end

Podstawowe twierdzenie kwadratur Gaussa¶

Odcięte xi N-punktowej kwadratury Gaussa z funkcją ważącą w(x) w przedziale [a,b] są zerami wielomianu ortogonalnego φ(x) dla tego samego przedziału i tej samej funkcji ważącej.

function calculate_integral_interval(f, k, a, b)
    (xp, w) = gauss(Float64, k)
    return (b-a)/2 * sum(w .* f.((b-a)/2 .* xp .+ (b+a)/2))
end

# n - liczba przedziałów

function trapezoidal(f, a, b, n)
    sum = 0;
    
    # Segment length
    h = (Float64);
    h = (b - a) / n;
    
    # f(a) + f(b)
    sum = f(a) + f(b);
    
    # Middle terms
    for i in 1:n-1
        sum += 2 * f(a + i * h);
    end
    
    return (h / 2) * sum;   
end
